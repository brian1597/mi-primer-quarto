datos <- read.xlsx ("C:\\Users\\Brian.Ruiz\\OneDrive - Universidad Tecnológica de Pereira\\1. UTP\\2. Introducción a la Ciencia de Datos\\2026-03-Trabajo Fina\\DATA\\Evaluaciones_Agropecuarias_Municipales_–_EVA._2019_-_2024._Base_Agrícola_20260327.xlsx")


datos <- datos %>% clean_names() # Limpieza de nombres de columnas
datos <- datos %>% select (-codigo_dane_departamento, -codigo_dane_municipio, -codigo_del_cultivo )
nombres_top3 <- datos %>% group_by(cultivo) %>%
  summarise(produccion_total = sum(produccion)) %>% 
  arrange(desc(produccion_total)) %>% 
  slice_head(n = 3) %>% # filtro los 3 cultivos de mayor producción
  pull(cultivo) # Saco los nombres de los 3 cultivos que graficaré :)
p  <- datos %>% 
  filter(cultivo %in% nombres_top3 ) %>% 
  group_by(ano, cultivo) %>%
  summarise(produccion_anual = round(sum(produccion)), .groups = "drop") %>% 
  
  ggplot(aes(x = ano, y = produccion_anual, color = cultivo)) +
  geom_line_interactive(aes(tooltip = cultivo, data_id = cultivo)) +
  geom_point_interactive(aes(tooltip = paste0("Año: ", ano, "\nProducción: ", comma(produccion_anual)))) +

  # 2. Corregimos el eje Y para que use puntos de miles
  scale_y_continuous(labels = label_comma(big.mark = ".", decimal.mark = ",")) +
  
  labs(
    title = str_wrap("Evolución de la Producción de los 3 cultivos principales de Colombia entre 2019 y 2024", width = 55), 
    x = "Año", 
    y = "Producción (Toneladas)"
  )+
  
  theme_minimal() 

girafe(ggobj = p)
# --- ANÁLISIS DE LA VARIABLE DEPARTAMENTO ---

# 1. Crear la tabla de frecuencia (Top 10)
tabla_dept <- datos %>%
  count(departamento, name = "cantidad") %>%
  arrange(desc(cantidad)) %>%
  slice_head(n = 10)

# 2. Guardar variables para el texto automatizado
top_dept <- tabla_dept$departamento[1]
num_dept <- comma(tabla_dept$cantidad[1], big.mark = ".", decimal.mark = ",")

# 3. Crear la gráfica de barras interactiva
p_dept <- tabla_dept %>%
  ggplot(aes(x = reorder(departamento, cantidad), y = cantidad, fill = departamento)) +
  geom_col_interactive(aes(tooltip = paste0(departamento, ": ", comma(cantidad, big.mark = ".", decimal.mark = ",")), data_id = departamento), 
                       show.legend = FALSE) + # Ocultamos leyenda porque los nombres ya están en el eje
  coord_flip() + # Giramos la gráfica para leer mejor los nombres
  scale_y_continuous(labels = label_comma(big.mark = ".", decimal.mark = ",")) +
  labs(
    title = "Top 10 Departamentos con mayor número de registros agrícolas",
    x = "Departamento",
    y = "Cantidad de Registros"
  ) +
  theme_minimal()

girafe(ggobj = p_dept)

De acuerdo con la tabla de frecuencias y la gráfica anterior, el departamento que cuenta con la mayor cantidad de registros en las evaluaciones agropecuarias es Boyacá, acumulando un total de 14.969 registros.
