# Usar archivo local
datos <- read.xlsx("DATA/Evaluaciones_Agropecuarias_Municipales_–_EVA._2019_-_2024._Base_Agrícola_20260327.xlsx")


datos <- datos %>% clean_names() # Limpieza de nombres de columnas
datos <- datos %>% select(-codigo_dane_departamento, -codigo_dane_municipio, -codigo_del_cultivo)
nombres_top3 <- datos %>% group_by(cultivo) %>%
  summarise(produccion_total = sum(produccion)) %>% 
  arrange(desc(produccion_total)) %>% 
  slice_head(n = 3) %>% # filtro los 3 cultivos de mayor producción
  pull(cultivo) # Saco los nombres de los 3 cultivos que graficaré :)
p  <- datos %>% 
  filter(cultivo %in% nombres_top3 ) %>% 
  group_by(ano, cultivo) %>%
  summarise(produccion_anual = round(sum(produccion)), .groups = "drop") %>% 
  
  ggplot(aes(x = ano, y = produccion_anual, color = cultivo)) +
  geom_line_interactive(aes(tooltip = cultivo, data_id = cultivo)) +
  geom_point_interactive(aes(tooltip = paste0("Año: ", ano, "\nProducción: ", comma(produccion_anual)))) +

  # 2. Corregimos el eje Y para que use puntos de miles
  scale_y_continuous(labels = label_comma(big.mark = ".", decimal.mark = ",")) +
  
  labs(
    title = str_wrap("Evolución de la Producción de los 3 cultivos principales de Colombia entre 2019 y 2024", width = 55), 
    x = "Año", 
    y = "Producción (Toneladas)"
  )+
  
  theme_minimal() 

girafe(ggobj = p)
# --- ANÁLISIS DE LA VARIABLE DEPARTAMENTO ---

# 1. Crear la tabla de frecuencia (Top 10)
tabla_dept <- datos %>%
  count(departamento, name = "cantidad") %>%
  arrange(desc(cantidad)) %>%
  slice_head(n = 10)

# 2. Guardar variables para el texto automatizado
top_dept <- tabla_dept$departamento[1]

# 3. Crear la gráfica de barras interactiva
p_dept <- tabla_dept %>%

                       show.legend = FALSE) + # Ocultamos leyenda porque los nombres ya están en el eje
  coord_flip() + # Giramos la gráfica para leer mejor los nombres
  scale_y_continuous(labels = label_comma(big.mark = ".", decimal.mark = ",")) +
  labs(
    title = "Top 10 Departamentos con mayor número de registros agrícolas",
    x = "Departamento",
    y = "Cantidad de Registros"
  ) +
  theme_minimal()

girafe(ggobj = p_dept)

De acuerdo con la tabla de frecuencias y la gráfica anterior, el departamento que cuenta con la mayor cantidad de registros en las evaluaciones agropecuarias es Boyacá, acumulando un total de 14.969 registros.

matriz_cor <- cor(var_numericas)

# 4. Crear gráfica de correlación con ggcorrplot
p_cor <- ggcorrplot(matriz_cor, 
                     method = "circle", 
                     type = "lower",
                     lab = TRUE,
                     lab_size = 4,
                     colors = c("#d73027", "#ffffbf", "#1a9850"),
                     title = "Matriz de Correlación entre Variables Numéricas",
                     ggtheme = theme_minimal())

girafe(ggobj = p_cor)

La matriz de correlación revela que existe una correlación positiva fuerte de 0.436 entre el área sembrada y la producción, mientras que la relación entre producción y rendimiento es de 0.248, lo cual indica una asociación directa entre estas variables del sector agropecuario.

# --- PASO 4: MEDIDAS RESUMEN POR CULTIVO ---

# 1. Calcular medidas resumen para los top 3 cultivos
medidas_resumen <- datos %>%
  filter(cultivo %in% nombres_top3) %>%
  group_by(cultivo) %>%
  summarise(
    Min = min(produccion, na.rm = TRUE),
    Q1 = quantile(produccion, 0.25, na.rm = TRUE),
    Mediana = median(produccion, na.rm = TRUE),
    Media = round(mean(produccion, na.rm = TRUE), 2),
    Q3 = quantile(produccion, 0.75, na.rm = TRUE),
    Max = max(produccion, na.rm = TRUE),
    IQR = IQR(produccion, na.rm = TRUE),
    DE = round(sd(produccion, na.rm = TRUE), 2),
    .groups = "drop"
  )

# Mostrar tabla
datatable(medidas_resumen, options = list(pageLength = 5))
# 2. Variables para texto automatizado
cultivo_max_media <- medidas_resumen %>% slice_max(Media, n = 1) %>% pull(cultivo)
valor_max_media <- comma(medidas_resumen %>% slice_max(Media, n = 1) %>% pull(Media), big.mark = ".", decimal.mark = ",")
cultivo_max_iqr <- medidas_resumen %>% slice_max(IQR, n = 1) %>% pull(cultivo)
valor_max_iqr <- comma(medidas_resumen %>% slice_max(IQR, n = 1) %>% pull(IQR), big.mark = ".", decimal.mark = ",")

El análisis de medidas resumen por cultivo muestra que Caña presenta la mayor producción promedio con 65.525 toneladas. En cuanto a la variabilidad, Caña tiene el mayor rango intercuartílico con 19.620 toneladas, indicando una mayor dispersión en los datos de producción.

# 6. Prueba de normalidad (Shapiro-Wilk)
muestra_prod <- datos %>% 
  filter(cultivo == cultivo_ejemplo_dist, !is.na(produccion)) %>% 
  pull(produccion) %>% 
  head(5000)

shapiro_test <- shapiro.test(muestra_prod)
estadistico_shap <- round(shapiro_test$statistic, 4)
valor_p_shap <- format(shapiro_test$p.value, scientific = TRUE)
normalidad <- ifelse(valor_p_shap < 0.05, "NO se ajusta a una distribución normal", "se ajusta a una distribución normal")

Los histogramas, gráficos de densidad y distribución empírica muestran la forma de la variable producción para el cultivo de Caña. La prueba de Shapiro-Wilk obtiene un estadístico de 0.2237 con un valor-p de 1.914526e-84, lo que indica que los datos se ajusta a una distribución normal (al nivel de significancia 0.05).

# --- PASO 6: INTERVALOS DE CONFIANZA ---

# 1. Intervalo de confianza para la media de producción del cultivo #1
